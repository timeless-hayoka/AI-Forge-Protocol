New Function (in dmu.py — apply_dynamic_diversity_floor):
PythonCopydef get_dynamic_diversity_cap(
    energy: float = 0.5,
    shadow_depth: float = 0.3,
    pedi_stability: float = 0.7,
    psc_chaos: float = 0.2,
    homeostasis_stress: float = 0.4,
) -> float:
    """
    Adaptive cap: tighter under stress (prevent loops), looser during integration.
    """
    base = 0.40

    # Stress signals tighten the floor
    stress_factor = (
        (1.0 - energy) * 0.25 +
        shadow_depth * 0.20 +
        (1.0 - pedi_stability) * 0.15 +
        psc_chaos * 0.25 +
        homeostasis_stress * 0.15

    # Calm / integration signals loosen it
    integration_factor = energy * 0.1 + pedi_stability * 0.15

    cap = base - stress_factor + integration_factor

textCopyclass DriftSystem:
    def __init__(self):
        self.memory = {"signal": 1.0, "noise": 0.1}

    def measure_deviation(self, baseline, current):
        deviation = {}
        for k in baseline:
            b = baseline[k]
            c = current.get(k, 0)

            deviation[k] = abs(b - c) if c is not None else 1.0

        return deviation

    def classify_deviation(self, deviation):
        score = sum(deviation.values())

        if score < 0.5:
            return "noise"
        elif score < 2.0:
            return "drift"
        else:
            return "anomaly"

    def apply_correction(self, current, classification):
        corrected = current.copy()

        if classification == "noise":
            factor = 0.1
        elif classification == "drift":
            factor = 0.3
        else:
            factor = 0.6

        for k in corrected:
            if corrected[k] is not None:
                corrected[k] = corrected[k] * (1 - factor)

        return corrected

    def update_memory(self, corrected):
        self.memory = corrected

textCopydef evaluate(baseline, current, corrected):
    def error(a, b):
        return sum(abs(a[k] - b.get(k, 0)) for k in a)

    before = error(baseline, current)
    after = error(baseline, corrected)

    return {
        "error_before": before,
        "error_after": after,
        "improvement": before - after
    }

    return logs

class TelemetryLogger:
    def __init__(self, path: Path = TELEMETRY_PATH):
        self.path = path

    def log(self, data: dict):
        data["timestamp"] = datetime.utcnow().isoformat()
        with open(self.path, "a", encoding="utf-8") as f:
            f.write(json.dumps(data, default=str) + "\n")


class ExceptionLogger:
    def __init__(self, path: Path = EXCEPTIONS_PATH):
        self.path = path

    def log_exception(self, exc: Exception, context: dict = None):
        error_data = {
            "type": type(exc).__name__,
            "message": str(exc),
            "traceback": traceback.format_exc(),
            "context": context or {},
            "timestamp": datetime.utcnow().isoformat()
        }
        with open(self.path, "a", encoding="utf-8") as f:
            f.write(json.dumps(error_data, default=str) + "\n")
        console.print(f"[red]Exception logged to {self.path}[/red]")
