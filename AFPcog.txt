def afp_estimate_complexity(user_input: str, context: dict = None) -> float:
    """AFP Complexity Score: 0.0 (trivial) → 1.0 (Forge-tier stress)"""
    score = 0.0
    text = user_input.lower()
    length = len(user_input)
    
    # Length & structural
    if length > 300: score += 0.25
    elif length > 150: score += 0.12
    
    # Semantic triggers (why/how/design/error/architecture/shadow/reflect)
    triggers = ["why", "how", "design", "architecture", "error", "debug", "shadow", "reflect", "dissonance", "prune", "unify"]
    score += 0.18 * sum(1 for t in triggers if t in text)
    
    # Ambiguity / multi-intent
    if len(re.findall(r'\b(and|or|both|vs|compare|integrate)\b', text)) > 2:
        score += 0.15
    
    # Context pressure (from causal_wiring / homeostasis)
    if context:
        pedi = context.get("pedi", {})
        homeo = context.get("homeostasis", {})
        score += 0.2 * (1 - pedi.get("coherence", 0.5))
        score += 0.15 * homeo.get("stress", 0.0)
    
    return min(max(score, 0.0), 1.0)

def afp_select_strategy(score: float) -> Tuple[str, dict]:
