The Universal Adapter Contract

Any system that implements this simple interface can be tested by TestForge. No rewrites. No lock-in.

class TestForgeAdapter:
    def generate(self, prompt: str, context: dict = None) -> dict:
        """
        Returns:
        {
            "output": str,           # The generated response
            "trace": list[str],      # Optional action trace
            "state": dict,           # Optional internal state snapshot
            "metrics": dict          # Optional native metrics
        }
        """
        raise NotImplementedError
